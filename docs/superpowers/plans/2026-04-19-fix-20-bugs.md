```python
async def analyze(self, state: RequirementAnalysisState) -> Dict[str, Any]:
```

## Task 2: Fix PCM Decoding in transcribe_stream (BUG-10)

The current code reads raw PCM bytes as `float32`, which misinterprets int16 samples (halves the count). Then passes raw bytes to Whisper, which expects a decoded audio file or a float32 numpy array.

**Files:**
- Modify: `backend/core/transcriber.py`

- [ ] **Step 1: Fix `transcribe_stream` to decode int16 PCM correctly**

Replace the entire `transcribe_stream` method body in `backend/core/transcriber.py`:

```python
def transcribe_stream(self, audio_chunk: bytes, chunk_number: int = 0) -> Optional[str]:
    """Transcribe a single audio chunk from a stream (raw int16 PCM at 16kHz)."""
    try:
        # Raw PCM from browser MediaRecorder is int16. Interpret correctly.
        audio_data = np.frombuffer(audio_chunk, dtype=np.int16).astype(np.float32) / 32768.0

        # Minimum 500ms of audio at 16kHz = 8000 samples
        if len(audio_data) < 8000:
            logger.debug(f"Chunk {chunk_number} too short ({len(audio_data)} samples), skipping")
            return None

        segments, _ = self.model.transcribe(
            audio_data,
            language="en",
            beam_size=3,
            temperature=0.0,
        )

        text = " ".join([segment.text for segment in segments]).strip()
        if text:
            logger.info(f"Stream chunk {chunk_number} transcribed: {len(text)} chars")
            return text
        return None

    except Exception as e:
        logger.warning(f"Stream transcription error on chunk {chunk_number}: {str(e)}")
        return None
```

```python
async def _handle_audio_chunk(
    self, session_id: str, message: Dict, context_docs: List, accumulated_text: str
) -> str:
    """Handle incoming audio chunk. Returns updated accumulated_text."""
    try:
        chunk_data = message.get("data")
        chunk_number = message.get("chunk_number", 0)

        if isinstance(chunk_data, str):
            audio_bytes = base64.b64decode(chunk_data)
        else:
            audio_bytes = chunk_data

        transcription = self.transcriber.transcribe_stream(audio_bytes, chunk_number)

        if transcription:
            accumulated_text = accumulated_text + " " + transcription

        await self.ws_manager.send_message(
            session_id,
            {
                "type": "chunk_ack",
                "chunk_number": chunk_number,
                "transcription": transcription or "",
                "accumulated_length": len(accumulated_text),
            },
        )

        # Trigger analysis every 50 characters of accumulated text
        if len(accumulated_text) > 0 and len(accumulated_text) % 50 == 0:
            logger.info(f"Triggering analysis for session {session_id} at {len(accumulated_text)} chars")
            await self._run_analysis(session_id, accumulated_text, context_docs)

    except Exception as e:
        logger.error(f"Audio chunk handling error: {str(e)}")
        await self.ws_manager.send_message(
            session_id,
            {"type": "error", "message": f"Audio chunk error: {str(e)}"},
        )

    return accumulated_text
```

## Task 6: Fix `_extract_score` Regex (BUG-19)

`re.search(r'\d+\.?\d*', text)` grabs the first digit found, which is typically from enumerated list items (`"1. Missing..."` → score=0.1). Fix: look for a labeled float first.

**Files:**
- Modify: `backend/core/agents.py`

- [ ] **Step 1: Replace `_extract_score` method**

Find and replace the entire `_extract_score` method in `agents.py`:

```python
def _extract_score(self, text: str) -> float:
    """Extract a 0-1 score from LLM response, looking for labeled values first."""
    import re
    try:
        # Prefer labeled patterns: "score: 0.7", "consistency: 0.85", "gap: 0.3"
        labeled = re.search(
            r'(?:score|gap|consistency|rating|quality)[\s:=]+([01](?:\.\d+)?)',
            text,
            re.IGNORECASE,
        )
        if labeled:
            return float(labeled.group(1))

        # Fall back: find a standalone decimal like "0.72" or "0.8"
        decimal = re.search(r'\b0\.\d+\b', text)
        if decimal:
            return float(decimal.group())

        return 0.5
    except Exception:
        return 0.5
```

```python
@router.post("/formalize", response_model=FormalizedRequirement)
async def formalize_requirements(session_id: str):
    try:
        logger.info(f"Retrieving requirements from memory for session {session_id}")

        agent = get_agent()
        config = {"configurable": {"thread_id": session_id}}
        state = agent.app.get_state(config)

        if not state.values:
            raise HTTPException(status_code=404, detail="No data found for this session ID")

        # Only return formalized data if the workflow has reached a terminal state
        current_status = state.values.get("status", "")
        if current_status not in ("formal_draft", "export_ready", "clarified"):
            raise HTTPException(
                status_code=409,
                detail=f"Analysis still in progress (status={current_status}). Retry after analysis completes."
            )

        raw_requirements = state.values.get("requirements", [])

        # Coerce dicts → ISORequirement, filling in defaults for missing fields
        iso_requirements = []
        for i, req in enumerate(raw_requirements, 1):
            if not isinstance(req, dict):
                continue
            iso_requirements.append({
                "requirement_id": req.get("requirement_id", f"REQ-{i:04d}"),
                "title": req.get("title", "Untitled Requirement"),
                "shall_statement": req.get("shall_statement", "The system shall meet this requirement."),
                "rationale": req.get("rationale", ""),
                "acceptance_criteria": req.get("acceptance_criteria") or [],
                "priority": req.get("priority", "Medium"),
                "category": req.get("category"),
                "traceability": req.get("traceability") or req.get("depends_on") or [],
            })

        completeness = state.values.get("formalized", {}).get("completeness_score", 0.0)
        summary = f"Found {len(iso_requirements)} ISO 29148 requirements"

        return FormalizedRequirement(
            iso_requirements=iso_requirements,
            summary=summary,
            total_requirements=len(iso_requirements),
            completeness_score=completeness,
            ready_for_export=len(iso_requirements) > 0,
            export_formats=["jira", "trello"],
        )

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Formalization error: {str(e)}")
        raise HTTPException(status_code=500, detail=str(e))
```

```python
async def analyze_requirements(request: AnalyzeRequirementsRequest):

        gpu_manager = get_gpu_manager()

            agent = get_agent()
            result = await agent.analyze(state)

        raise HTTPException(status_code=500, detail=str(e))
```

```python
    except asyncio.CancelledError:
        pass
```

## Task 16: Add Basic WebSocket Connection Lock (BUG-13)

`WebSocketManager.active_connections` is a plain dict. Between `await` points in `send_message`/`receive_message`, concurrent sessions can see torn state. Add an `asyncio.Lock` to guard the dict.

**Files:**
- Modify: `backend/api/websocket_handler.py`

- [ ] **Step 1: Add a lock to `WebSocketManager.__init__`**

```python
def __init__(self):
    self.active_connections: Dict[str, WebSocket] = {}
    self._lock = asyncio.Lock()
    logger.info("WebSocket Manager initialized")
```

- [ ] **Step 2: Guard `connect` and `disconnect` with the lock**

```python
async def connect(self, websocket: WebSocket, session_id: str) -> None:
    await websocket.accept()
    async with self._lock:
        self.active_connections[session_id] = websocket
    logger.info(f"WebSocket connected for session {session_id}")

async def disconnect(self, session_id: str) -> None:
    async with self._lock:
        self.active_connections.pop(session_id, None)
    logger.info(f"WebSocket disconnected for session {session_id}")
```
